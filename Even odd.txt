import java.util.*;

public class main {
public static void main(String[] args) {
    Scanner scn = new Scanner(System.in);
    int x = scn.nextInt();
    if (x % 2 == 0)
    {
      System.out.print (x + "is even");
    }
    else
    {
      System.out.print (x + "is odd");
    }
  }
}
